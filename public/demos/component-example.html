Map Configuration

Example 1: Dynamic Map

üìù HTML Markup

<philippines-map
    id="mainMap"
    level="1"
    title="Philippines Regions"
    value-name="Random Value"
    color-scheme="blue"
    height="600px">
</philippines-map>

<script>
const mainMap = document.getElementById('mainMap');

// Listen to events
mainMap.addEventListener('region-click', (event) => {
    const feature = event.detail.feature;
    console.log('Clicked:', feature.ADM1_EN);
});

mainMap.addEventListener('map-ready', (event) => {
    console.log('Map ready with', event.detail.features.length, 'features');
});

// Update map attributes dynamically
mainMap.setAttribute('level', '2');
mainMap.setAttribute('color-scheme', 'green');

// Load custom data
mainMap.setData({
    'PH01': 85.5,
    'PH02': 72.3,
    'PH03': 91.2
});
</script>
Example 2: All Provinces (Green)

üìù HTML Markup

<philippines-map
    level="2"
    title="All Provinces"
    value-name="Sample Data"
    color-scheme="green"
    height="400px">
</philippines-map>

<script>
// Province-level example (Level 2)
// Uses combined file: provdists-all.0.001.json
// Contains all 88 provinces/districts

const provinceMap = document.querySelector('philippines-map[level="2"]');

// Load province-specific data
provinceMap.addEventListener('map-ready', () => {
    provinceMap.setData({
        'PH01028': 75,  // Ilocos Norte
        'PH01029': 82,  // Ilocos Sur
        'PH02009': 68   // Batanes
    });
});
</script>
