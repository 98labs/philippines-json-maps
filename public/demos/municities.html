Interactive demo showing barangays by municipality with resolution selector

// Initialize map centered on Philippines
var map = L.map('map').setView([12.8797, 121.7740], 6);

// Add OpenStreetMap tiles
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap contributors',
    maxZoom: 19
}).addTo(map);

// Color function based on value (choropleth)
function getColor(d) {
    return d > 80 ? '#00429d' :
           d > 70 ? '#2e59a8' :
           d > 60 ? '#4771b2' :
           d > 50 ? '#5d8abd' :
           d > 40 ? '#73a2c6' :
           d > 30 ? '#8abccf' :
           d > 20 ? '#a5d5d8' :
                    '#c1e7e3';
}

// Style function for each feature
function style(feature) {
    return {
        fillColor: getColor(feature.properties.randomValue),
        weight: 1.5,
        opacity: 1,
        color: 'white',
        dashArray: '2',
        fillOpacity: 0.7
    };
}

// Resolution mapping
var currentResolution = 'lowres';
var resolutionMap = {
    'lowres': { suffix: '0.001', folder: 'lowres' },
    'medres': { suffix: '0.01', folder: 'medres' },
    'hires': { suffix: '0.1', folder: 'hires' }
};

// Load municipality data
function loadMunicipality(municipalityCode) {
    var config = resolutionMap[currentResolution];
    var url = `../2023/geojson/municities/${config.folder}/bgysubmuns-municity-${municipalityCode}.${config.suffix}.json`;

    fetch(url)
        .then(response => response.json())
        .then(data => {
            // Assign random values to each feature
            data.features.forEach(feature => {
                feature.properties.randomValue = Math.floor(Math.random() * 100);
            });

            // Add to map
            if (geojsonLayer) {
                map.removeLayer(geojsonLayer);
            }
            geojsonLayer = L.geoJson(data, {
                style: style,
                onEachFeature: function(feature, layer) {
                    layer.on({
                        mouseover: highlightFeature,
                        mouseout: resetHighlight,
                        click: zoomToFeature
                    });
                }
            }).addTo(map);

            map.fitBounds(geojsonLayer.getBounds());
        });
}
