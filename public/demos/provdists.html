Interactive demo showing all 1,642 Philippine municipalities/cities with resolution selector

// Initialize map centered on Philippines
        var map = L.map('map').setView([12.8797, 121.7740], 6);

        // Add OpenStreetMap tiles
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors',
            maxZoom: 19
        }).addTo(map);

        // Color function based on value
        function getColor(d) {
            return d > 80 ? '#4A148C' :
                   d > 70 ? '#6A1B9A' :
                   d > 60 ? '#8E24AA' :
                   d > 50 ? '#AB47BC' :
                   d > 40 ? '#BA68C8' :
                   d > 30 ? '#CE93D8' :
                   d > 20 ? '#E1BEE7' :
                            '#F3E5F5';
        }

        // Style function
        function style(feature) {
            return {
                fillColor: getColor(feature.properties.randomValue),
                weight: 1,
                opacity: 1,
                color: 'white',
                dashArray: '2',
                fillOpacity: 0.7
            };
        }

        // Highlight feature on hover
        function highlightFeature(e) {
            var layer = e.target;

            layer.setStyle({
                weight: 3,
                color: '#666',
                dashArray: '',
                fillOpacity: 0.7
            });

            layer.bringToFront();
            info.update(layer.feature.properties);
        }

        var geojsonLayer;

        // Reset highlight
        function resetHighlight(e) {
            if (geojsonLayer) {
                geojsonLayer.resetStyle(e.target);
            }
            info.update();
        }

        // Zoom to feature on click
        function zoomToFeature(e) {
            map.fitBounds(e.target.getBounds());
        }

        // Attach events to each feature
        function onEachFeature(feature, layer) {
            layer.on({
                mouseover: highlightFeature,
                mouseout: resetHighlight,
                click: zoomToFeature
            });
        }

        var currentResolution = 'lowres';

        // Resolution to file suffix mapping
        var resolutionMap = {
            'lowres': '0.001',
            'medres': '0.01',
            'hires': '0.1'
        };

        // Load all municipalities data based on resolution
        function loadData(resolution) {
            var suffix = resolutionMap[resolution];
            var url = `../2023/municities-all.${suffix}.json`;

            // Remove existing layer if present
            if (geojsonLayer) {
                map.removeLayer(geojsonLayer);
            }

            fetch(url)
                .then(response => response.json())
                .then(data => {
                    // Assign random values to each feature
                    data.features.forEach(feature => {
                        feature.properties.randomValue = Math.floor(Math.random() * 100);
                    });

                    geojsonLayer = L.geoJson(data, {
                        style: style,
                        onEachFeature: onEachFeature
                    }).addTo(map);

                    // Fit map to data bounds
                    map.fitBounds(geojsonLayer.getBounds());
                })
                .catch(error => console.error('Error loading GeoJSON:', error));
        }

        // Initial load
        loadData(currentResolution);

        // Info control
        var info = L.control();

        info.onAdd = function (map) {
            this._div = L.DomUtil.create('div', 'info');
            this.update();
            return this._div;
        };

        info.update = function (props) {
            this._div.innerHTML = '<h4>All Municipalities/Cities (1,642)</h4>' +
                (props ?
                '<b>' + props.ADM3_EN + '</b><br />' +
                'Random Value: ' + props.randomValue.toFixed(2) + '<br />' +
                'Area: ' + (props.AREA_SQKM ? props.AREA_SQKM.toFixed(2) : 'N/A') + ' km¬≤'
                : 'Hover over a municipality/city');
        };

        info.addTo(map);

        // Legend control
        var legend = L.control({position: 'bottomright'});

        legend.onAdd = function (map) {
            var div = L.DomUtil.create('div', 'info legend'),
                grades = [0, 20, 30, 40, 50, 60, 70, 80],
                labels = [];

            div.innerHTML = '<h4>Random Value</h4>';

            for (var i = 0; i < grades.length; i++) {
                div.innerHTML +=
                    '<i style="background:' + getColor(grades[i] + 1) + '"></i> ' +
                    grades[i] + (grades[i + 1] ? '&ndash;' + grades[i + 1] + '<br>' : '+');
            }

            return div;
        };

        legend.addTo(map);

        // Resolution control
        var resolutionControl = L.control({position: 'topleft'});

        resolutionControl.onAdd = function (map) {
            var div = L.DomUtil.create('div', 'resolution-control');
            div.innerHTML = `
                <label for="resolution-select">Resolution:</label>
                <select id="resolution-select">
                    <option value="lowres" selected>Low (fast)</option>
                    <option value="medres">Medium</option>
                    <option value="hires">High (detailed)</option>
                </select>
            `;

            // Prevent map clicks on the control
            L.DomEvent.disableClickPropagation(div);

            return div;
        };

        resolutionControl.addTo(map);

        // Add change event listener to resolution selector
        document.addEventListener('DOMContentLoaded', function() {
            setTimeout(function() {
                var selector = document.getElementById('resolution-select');
                if (selector) {
                    selector.addEventListener('change', function(e) {
                        currentResolution = e.target.value;
                        loadData(currentResolution);
                    });
                }
            }, 100);
        });
